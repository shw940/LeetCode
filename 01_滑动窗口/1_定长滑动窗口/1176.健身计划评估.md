
### 题目

[1176. 健身计划评估](https://leetcode.cn/problems/diet-plan-performance/)


### 题解



### 代码

```cpp
class Solution {
public:
    int dietPlanPerformance(vector<int>& calories, int k, int lower,
                            int upper) {
        int score = 0;
        int sum_calories = 0;

        for (int l = 0, r = 0; r < calories.size(); r++) {
            sum_calories += calories[r];
            if (r - l + 1 == k) {
                if (sum_calories < lower) {
                    score--;
                }
                if (sum_calories > upper) {
                    score++;
                }
                sum_calories -= calories[l++];
            }
        }

        return score;
    }
};
```
