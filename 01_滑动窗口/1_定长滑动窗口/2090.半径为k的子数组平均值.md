
### 题目: 

[2090. 半径为k的子数组平均值](https://leetcode.cn/problems/k-radius-subarray-averages/)


### 题解



### 代码

```cpp
class Solution {
public:
    vector<int> getAverages(vector<int>& nums, int k) {
        vector<int> average(nums.size(), -1);
        long long now_sum = 0;

        for (int l = 0, r = 0; r < nums.size(); r++) {
            now_sum += nums[r];
            if (r - l + 1 == 2 * k + 1) {
                average[r - k] = now_sum / (2 * k + 1);
                now_sum -= nums[l++];
            }
        }

        return average;
    }
};
```
