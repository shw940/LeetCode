
### 题目

[1852. 每个子数组的数字种类数](https://leetcode.cn/problems/distinct-numbers-in-each-subarray/)


### 题解



### 代码

```cpp
class Solution {
public:
    vector<int> distinctNumbers(vector<int>& nums, int k) {
        vector<int> ans;
        unordered_map<int, int> mp;

        for (int l = 0, r = 0; r < nums.size(); r++) {
            mp[nums[r]]++;
            if (r - l + 1 == k) {
                ans.push_back(mp.size());
                if (--mp[nums[l]] == 0) {
                    mp.erase(nums[l]);
                }
                l++;
            }
        }

        return ans;
    }
};
```
