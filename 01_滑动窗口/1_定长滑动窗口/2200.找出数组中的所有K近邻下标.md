class Solution {
public:
    vector<int> findKDistantIndices(vector<int>& nums, int key, int k) {
        vector<int> ans;
        for (int l = 0, r = 0; r < nums.size(); r++) {
            if (nums[r] == key) {
                l = max(l, r - k);
                while (l < nums.size() && l - r <= k) {
                    ans.push_back(l++);
                }
            }
        }

        return ans;
    }
};
