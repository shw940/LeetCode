
### 题目

[1456. 定长子串中元音的最大数目](https://leetcode.cn/problems/maximum-number-of-vowels-in-a-substring-of-given-length/)


### 题解



### 代码

```cpp
class Solution {
public:
    int maxVowels(string s, int k)
    {
        int max_vowels = 0;
        int now_vowels = 0;
        
        auto judge = [&](char ch) -> bool {
            return (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u');
        };
        
        for (int l = 0, r = 0; r < s.size(); r++) {
            now_vowels += judge(s[r]);
            if (r - l + 1 == k) {
                max_vowels = max(max_vowels, now_vowels);
                now_vowels -= judge(s[l++]);
            }
        }
        return max_vowels;
    }
};
```
