
### 题目

[1423. 可获得的最大点数](https://leetcode.cn/problems/maximum-points-you-can-obtain-from-cards/)


### 题解



### 代码

```cpp
class Solution {
public:
    int maxScore(vector<int>& cardPoints, int k) {
        // 长度为 cardPoints.size() - k 的定长滑动窗口。
        // 求子数组的最小值

        long long min_point = 2e9;
        long long now_point = 0;
        long long sum_point = 0;

        for (int l = 0, r = 0; r < cardPoints.size(); r++) {
            now_point += cardPoints[r];
            sum_point += cardPoints[r];
            if (r - l + 1 == cardPoints.size() - k) {
                min_point = min(min_point, now_point);
                now_point -= cardPoints[l++];
            }
        }

        return min_point == (long long)2e9 ? sum_point : sum_point - min_point;
    }
};
```
