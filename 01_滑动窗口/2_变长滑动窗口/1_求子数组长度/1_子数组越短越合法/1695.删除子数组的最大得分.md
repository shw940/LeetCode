
### 题目

[1695. 删除子数组的最大得分](https://leetcode.cn/problems/maximum-erasure-value/)


### 题解



### 代码

```cpp
class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        int max_score = 0;
        int now_score = 0;
        vector<int> mp(1e4 + 10);
        int kinds = 0;

        for (int l = 0, r = 0; r < nums.size(); r++) {
            if (++mp[nums[r]] == 1) {
                kinds++;
            }
            now_score += nums[r];
            while (kinds != r - l + 1) {
                if (--mp[nums[l]] == 0) {
                    kinds--;
                }
                now_score -= nums[l++];
            }

            max_score = max(max_score, now_score);
        }
        
        return max_score;
    }
};
```
