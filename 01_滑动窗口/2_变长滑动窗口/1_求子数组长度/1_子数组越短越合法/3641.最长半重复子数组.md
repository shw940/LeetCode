
### 题目

[3641. 最长半重复子数组](https://leetcode.cn/problems/longest-semi-repeating-subarray/)


### 题解



### 代码

```cpp
class Solution {
public:
    int longestSubarray(vector<int>& nums, int k) {
        int max_len = 0;
        vector<int> mp(1e5 + 10);
        int cnt = 0;

        for (int l = 0, r = 0; r < nums.size(); r++) {
            if (++mp[nums[r]] == 2) {
                cnt++;
            }
            while (cnt == k + 1) {
                if (--mp[nums[l++]] == 1) {
                    cnt--;
                }
            }
            max_len = max(max_len, r - l + 1);
        }

        return max_len;
    }
};
```
