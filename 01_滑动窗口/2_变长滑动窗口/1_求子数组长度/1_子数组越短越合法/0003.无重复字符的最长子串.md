
### 题目

[3. 无重复字符的最长子串](https://leetcode.cn/problems/longest-substring-without-repeating-characters/)


### 题解



### 代码

```cpp
class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int max_len = 0;
        vector<int> mp(200);

        for (int l = 0, r = 0; r < s.size(); r++) {
            mp[s[r]]++;
            while (mp[s[r]] == 2) {
                mp[s[l++]]--;
            }
            max_len = max(max_len, r - l + 1);
        }

        return max_len;
    }
};
```
